\section{Pointers}
\frame{
	\frametitle{Pointers}
	
	\begin{itemize}
		\item declaration: type * name
		\item type is the data type being pointed to
		\item getting address of variable with \&
		\item dereferencing pointer with *
	\end{itemize}
For example:\\
int i = 10;\\
int *p = \&i;\\
*p = 15;
}
