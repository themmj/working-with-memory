\section{Header Files}
\frame{
	\frametitle{Header Files}
	\begin{itemize}
		\item file extension .h
		\item counterpart to source files (.c)
		\item contain function and data declarations
		\item contain macro definitions
		\item meant to be shared between multiple source files using \#include "name-of-header.h"
		\item are copied into source files $\rightarrow$ macro guards
	\end{itemize}
}
